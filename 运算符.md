## 1.加减乘除
```
#include <iostream>
using namespace std;
int main()
{
    int a = 10;
    int b = 3;
    cout << "a + b = " << a + b << endl;
    cout << "a - b = " << a - b << endl;
    cout << "a * b = " << a * b << endl;
    cout << "a / b = " << a / b << endl; // 两个整数相除，结果取整

    int c = 10;
    int d = 20;
    cout << "c / d = " << c / d << endl; // 两个整数相除，结果取整

    double e = 10.0;
    double f = 3.0;
    cout << "e / f = " << e / f << endl; // 两个浮点数相除，结果可为小数

    double m = 20.0;
    double n = 10.0;
    cout << "m / n = " << m / n << endl; // 两个浮点数相除，结果可为整数

    return 0;
}
```
>注意：除数不可为0
## 2.取余运算
```
#include <iostream>
using namespace std;
int main()
{
    int a = 10;
    int b = 3;
    cout << "a % b = " << a % b << endl; // 取模运算，结果为1

    int c = 10;
    int d = 20;
    cout << "c % d = " << c % d << endl; // 取模运算，结果为10

    return 0;
}
```
>注意：取模运算除数不可为0，且小数不可进行取余运算
## 3.递增递减
```
#include <iostream>
using namespace std;
int main()
{
    // 前置递增
    int a = 10;
    ++a; // 让变量a自增1
    cout << "a = " << a << endl;
    // 后置递增
    int b = 10;
    b++; // 让变量b自增1
    cout << "b = " << b << endl;
    // 前置递增
    int m = 10;
    int n = ++m; // 先让m自增1，然后进行表达式运算，将m的值赋给n
    cout << "m = " << m << ", n = " << n << endl;
    // 后置递增
    int x = 10;
    int y = x++; // 先进行表达式运算，将x的值赋给y，然后让x自增1
    cout << "x = " << x << ", y = " << y << endl;

    return 0;
}
```
